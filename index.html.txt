    <div class="container">
        <h1>📌 旅途冰箱贴</h1>
        <p class="desc">扫码留下你的旅途感想，再次扫码即可聆听</p>
        
        <button class="btn record-btn" id="recordBtn">🔴 开始录音</button>
        <button class="btn stop-btn" id="stopBtn" disabled>⏹️ 停止录音</button>
        <button class="btn play-btn" id="playBtn" disabled>▶️ 播放我的录音</button>
        
        <p class="tips" id="tips">每个用户自动拥有独立录音空间，隐私安全</p>
    </div>

    <script>
        // 核心：用浏览器唯一标识区分每个用户（固定二维码，不同用户独立录音）
        const getUserId = () => {
            let userId = localStorage.getItem('fridge_user_id');
            if (!userId) {
                userId = 'user_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
                localStorage.setItem('fridge_user_id', userId);
            }
            return userId;
        };

        const userId = getUserId();
        const recordBtn = document.getElementById('recordBtn');
        const stopBtn = document.getElementById('stopBtn');
        const playBtn = document.getElementById('playBtn');
        const tips = document.getElementById('tips');

        let mediaRecorder;
        let audioChunks = [];
        let audioUrl;

        // 页面加载时：自动加载当前用户的录音
        window.addEventListener('load', () => {
            const savedAudio = localStorage.getItem(`fridge_audio_${userId}`);
            if (savedAudio) {
                audioUrl = savedAudio;
                playBtn.disabled = false;
                tips.textContent = '✅ 已找到你的录音，点击播放聆听旅途回忆';
            }
        });

        // 1. 开始录音
        recordBtn.addEventListener('click', async () => {
            try {
                const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                mediaRecorder = new MediaRecorder(stream);
                audioChunks = [];

                mediaRecorder.addEventListener('dataavailable', (e) => {
                    audioChunks.push(e.data);
                });

                mediaRecorder.addEventListener('stop', () => {
                    const audioBlob = new Blob(audioChunks, { type: 'audio/wav' });
                    audioUrl = URL.createObjectURL(audioBlob);
                    
                    // 保存录音到用户专属空间
                    localStorage.setItem(`fridge_audio_${userId}`, audioUrl);
                    playBtn.disabled = false;
                    tips.textContent = '✅ 录音保存成功！再次扫码也能播放';
                });

                mediaRecorder.start();
                recordBtn.disabled = true;
                stopBtn.disabled = false;
                tips.textContent = '🎙️ 正在录音中...点击停止结束';
            } catch (err) {
                tips.textContent = '❌ 请允许麦克风权限，否则无法录音';
            }
        });

        // 2. 停止录音
        stopBtn.addEventListener('click', () => {
            mediaRecorder.stop();
            recordBtn.disabled = false;
            stopBtn.disabled = true;
        });

        // 3. 播放录音
        playBtn.addEventListener('click', () => {
            const audio = new Audio(audioUrl);
            audio.play();
            tips.textContent = '🔊 正在播放你的旅途感想...';
        });
    </script>
